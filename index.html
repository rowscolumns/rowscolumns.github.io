import { Layout, Grid } from 'rowscolumns/react';

export default function Dashboard() {
return (
<Layout
layout={Grid.col(lg, sm)}
layout-md={Grid.row(50, 50)}
>
<div> Sidebar </div>
<div> Main Content </div>
</Layout>
);
}
